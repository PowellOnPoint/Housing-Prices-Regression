1 Introduction

This analysis focuses on building predictive models for house prices using linear regression techniques, starting with simple linear regression and progressing to more complex multiple linear regression models.

1.1 Simple Linear Regression

Taking a simple approach first, we will model log(SalePrice) as a function of Above grade (ground) living area square feet (GrLivArea). GrLivArea is a strong predictor of SalePrice and is commonly used in housing price models.

636 is a 3395 sqft house sold for 200,000 as a abnormal sale. This is a large house sold for a very low price. Since abnormal sales can be anything from forclosure to short sale or trade, we may consider excluding this group. For this model we will only exclude the one outlier.

Only explaining another 3% of the variance, this model is not a significant improvement over the SLR. The residual plots show similar patterns to the SLR, with some improvement in homoscedasticity but still some curvature in the residuals vs fitted plot. The histogram of residuals looks more normal.

# Ordered evaluation of all categorical predictors by RSS drop
evaluate_all_categorical <- function() {
  base_formula <- "log(SalePrice) ~ GrLivArea + GarageArea + TotalBsmtSF + FullBath"

    full_formula <- as.formula(paste(base_formula, "+", var_name))
    
    reduced <- lm(as.formula(base_formula), data = train_filtered)
    full    <- lm(full_formula, data = train_filtered)
    
    anova_result <- anova(reduced, full)
    
    tibble(
      Variable   = var_name,
      `RSS Drop` = round(anova_result$`Sum of Sq`[2], 4),

      Levels     = length(unique(train_filtered[[var_name]]))
    )
  }) |> 
    arrange(desc(`RSS Drop`))
  
  print(results, n = Inf)

    }

  na_count <- sum(is.na(saleprice))
  if (na_count > 0) {

      Variable   = var_name,
      `RSS Drop` = round(anova_result$`Sum of Sq`[2], 4),
      `Adj R²`   = round(summary(full)$adj.r.squared, 4),
      `p-value`  = format.pval(anova_result$`Pr(>F)`[2], digits = 3),
      Levels     = length(unique(train_filtered[[var_name]]))
    )
  }) |> 
    arrange(desc(`RSS Drop`))
  
  print(results, n = Inf)

}

create_kaggle_submission <- function(model, filename = "submission.csv") {
  
  # Align factor levels to prevent "new levels" error
  test_processed <- test
  for (var in names(model$xlevels)) {
    if (var %in% names(test_processed)) {
      test_processed[[var]] <- factor(test_processed[[var]], 
                                      levels = model$xlevels[[var]])
    }
  }
  
  # Generate predictions on log scale
  test_pred <- predict(model, newdata = test_processed, interval = "none")
  
  # Back-transform to original scale
  saleprice <- exp(test_pred)
  
  # Count and replace any NAs with median
  na_count <- sum(is.na(saleprice))
  if (na_count > 0) {
    median_price <- median(saleprice, na.rm = TRUE)
    saleprice[is.na(saleprice)] <- median_price
    cat("→ Replaced", na_count, "NA predictions with median SalePrice =", 
        round(median_price, 2), "\n")
  }
  
  # Create submission in exact format as sample_submission.csv
  submission <- tibble(
    Id        = test$Id,
    SalePrice = as.double(saleprice)   # ensure dbl type
  )
  
  write_csv(submission, here("Data", filename))
  
  cat("Submission saved to: Data/", filename, "\n")
  cat("Rows:", nrow(submission), 
      " | Final NAs in SalePrice:", sum(is.na(submission$SalePrice)), "\n\n")
  
  invisible(na_count)
}
